1. In `src/domain/metro.rs`, add a `MetroActivity` enum:
   ```rust
   #[derive(Debug, Clone, PartialEq)]
   pub enum MetroActivity {
       Starting,
       Ready,
       Bundling { percent: Option<u8> },
       DeviceConnected,
       Error(String),
   }

   impl std::fmt::Display for MetroActivity {
       fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
           match self {
               Self::Starting => write!(f, "Starting..."),
               Self::Ready => write!(f, "Ready"),
               Self::Bundling { percent: Some(p) } => write!(f, "Bundling {}%", p),
               Self::Bundling { percent: None } => write!(f, "Bundling..."),
               Self::DeviceConnected => write!(f, "Device connected"),
               Self::Error(msg) => write!(f, "Error: {}", msg),
           }
       }
   }
   ```
   Add `pub activity: Option<MetroActivity>` field to `MetroManager`. Initialize to `None` in `new()`. Set to `None` in `clear()`. Set to `Some(MetroActivity::Starting)` in `set_starting()`.

In `render_worktree_table()` in `src/ui/panels.rs`, modify the row-building loop to insert a detail row after the metro-running worktree:

1. Change the `rows` construction from a simple `.map().collect()` to a loop that can insert extra rows:
   ```rust
   let mut rows: Vec<Row> = Vec::new();
   let mut metro_detail_row_indices: Vec<usize> = Vec::new(); // track which row indices are detail rows (for selection offset)

   for (i, wt) in state.worktrees.iter().enumerate() {
       // ... existing row building code (unchanged) ...
       rows.push(/* existing Row */);

       // If this worktree is running metro and we have activity info, add detail row
       if wt.metro_status == WorktreeMetroStatus::Running {
           if let Some(ref activity) = state.metro.activity {
               let activity_text = format!("  {} {}", "\u{2502}", activity); // vertical bar + activity
               let detail_row = Row::new(vec![
                   Cell::from(""),           // icon column empty
                   Cell::from(""),           // label column empty
                   Cell::from(Span::styled(
                       activity_text,
                       Style::default().fg(Color::Cyan),
                   )),
                   Cell::from(""),           // ticket column empty
                   Cell::from(""),           // dir column empty
               ])
               .style(Style::default().bg(Color::Rgb(0, 60, 0))); // match metro row bg
               metro_detail_row_indices.push(rows.len());
               rows.push(detail_row);
           }
       }
   }
   ```

2. The detail row is NOT selectable — it's purely visual. To handle this, adjust the TableState selected index to account for inserted detail rows. Before rendering the table, map the logical selection index to the visual row index:
   ```rust
   // Offset selected index to account for detail rows inserted before it
   if let Some(selected) = state.worktree_table_state.selected() {
       let mut visual_idx = selected;
       for &detail_idx in &metro_detail_row_indices {
           if detail_idx <= visual_idx {
               visual_idx += 1;
           }
       }
       state.worktree_table_state.select(Some(visual_idx));
   }
   ```

3. After rendering with `f.render_stateful_widget(table, area, &mut state.worktree_table_state)`, restore the logical selection:
   ```rust
   // Restore logical index after render
   if let Some(visual) = state.worktree_table_state.selected() {
       let mut logical = visual;
       for &detail_idx in metro_detail_row_indices.iter().rev() {
           if detail_idx <= logical {
               logical -= 1;
           }
       }
       // Only restore if it changed
       if logical != visual {
           state.worktree_table_state.select(Some(logical));
       }
   }
   ```
