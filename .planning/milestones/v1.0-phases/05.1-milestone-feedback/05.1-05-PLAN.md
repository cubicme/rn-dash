4. Rewire `Action::OpenClaudeCode` in update():
   - Replace the existing tmux-based implementation:
   ```rust
   Action::OpenClaudeCode => {
       if let Some(ref mux) = state.multiplexer {
           let wt = if !state.worktrees.is_empty() {
               let idx = state.worktree_table_state.selected().unwrap_or(0).min(state.worktrees.len() - 1);
               &state.worktrees[idx]
           } else { return; };

           let path = wt.path.clone();
           let name = format!("claude-{}", wt.path.file_name().and_then(|n| n.to_str()).unwrap_or("wt"));
           let flags = state.claude_flags.clone();
           let command = if flags.is_empty() {
               "claude".to_string()
           } else {
               format!("claude {}", flags)
           };

           // Clone the multiplexer trait object for the spawn — but we can't clone Box<dyn>.
           // Instead, since new_window is synchronous (std::process::Command), run it in a blocking spawn.
           // Actually, the simplest approach: call new_window directly since it's fast (just spawns a process).
           // But update() should stay non-blocking. Use tokio::task::spawn_blocking.
           let mux_is_tmux = format!("{:?}", mux).contains("TmuxAdapter");
           let mux_is_zellij = format!("{:?}", mux).contains("ZellijAdapter");

           tokio::task::spawn_blocking(move || {
               let result = if mux_is_tmux {
                   crate::infra::multiplexer::TmuxAdapter.new_window(&path, &name, &command)
               } else if mux_is_zellij {
                   crate::infra::multiplexer::ZellijAdapter.new_window(&path, &name, &command)
               } else {
                   Ok(())
               };
               if let Err(e) = result {
                   tracing::warn!("multiplexer new_window failed: {e}");
               }
           });
       }
   }
   ```

   NOTE: The trait object approach is awkward for spawn because Box<dyn Multiplexer> is not Clone. A simpler alternative: store a `multiplexer_type: MultiplexerType` enum alongside the trait object, and dispatch in the spawn. Or even simpler: since there are only two adapters, store an enum variant:

   Actually, the cleanest approach: add a `MultiplexerKind` enum to AppState:
   ```rust
   #[derive(Debug, Clone, Copy, PartialEq)]
   pub enum MultiplexerKind { Tmux, Zellij }
   ```
   And store `pub multiplexer_kind: Option<MultiplexerKind>` in AppState. Use this for dispatch in spawns. Keep the multiplexer trait for the trait boundary pattern but use the kind for actual dispatch when crossing async boundaries.

   Alternatively, simplest approach that works: just check env vars in the spawn:
   ```rust
   Action::OpenClaudeCode => {
       if state.multiplexer.is_none() { return; }
       let wt = /* get selected worktree */;
       let path = wt.path.clone();
       let name = format!("claude-{}", wt.path.file_name().and_then(|n| n.to_str()).unwrap_or("wt"));
       let command = if state.claude_flags.is_empty() {
           "claude".to_string()
       } else {
           format!("claude {}", state.claude_flags)
       };
       tokio::task::spawn_blocking(move || {
           // Re-detect inside spawn (env vars don't change mid-session)
           if let Some(mux) = crate::infra::multiplexer::detect_multiplexer() {
               if let Err(e) = mux.new_window(&path, &name, &command) {
                   tracing::warn!("multiplexer new_window failed: {e}");
               }
           }
       });
   }
   ```
   Use this simplest approach. The re-detection is cheap (2 env var checks) and avoids Clone issues.
