3. Add helper methods on AppState (or as free functions) to access per-worktree output:
   ```rust
   fn active_worktree_id(state: &AppState) -> Option<WorktreeId> {
       if state.worktrees.is_empty() { return None; }
       let idx = state.worktree_list_state.selected().unwrap_or(0)
           .min(state.worktrees.len() - 1);
       Some(state.worktrees[idx].id.clone())
   }

   fn active_output(state: &AppState) -> &VecDeque<String> {
       static EMPTY: std::sync::LazyLock<VecDeque<String>> = std::sync::LazyLock::new(VecDeque::new);
       if let Some(id) = active_worktree_id(state) {
           state.command_output_by_worktree.get(&id).unwrap_or(&EMPTY)
       } else {
           &EMPTY
       }
   }

   fn active_output_scroll(state: &AppState) -> usize {
       active_worktree_id(state)
           .and_then(|id| state.command_output_scroll_by_worktree.get(&id).copied())
           .unwrap_or(0)
   }
   ```
