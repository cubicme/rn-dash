   ```rust
       if let Some(ModalState::CleanToggle { options }) = state.modal.take() {
           state.palette_mode = None;

           // Build command sequence from checked options
           let mut cmds: Vec<CommandSpec> = Vec::new();
           if options.node_modules {
               cmds.push(CommandSpec::RmNodeModules);
           }
           if options.pods {
               cmds.push(CommandSpec::RnCleanCocoapods);
           }
           if options.android {
               cmds.push(CommandSpec::RnCleanAndroid);
           }
           if options.sync_after {
               cmds.push(CommandSpec::YarnInstall);
               cmds.push(CommandSpec::YarnPodInstall);
           }

           if cmds.is_empty() { return; }

           // Dispatch first, queue rest
           let first = cmds.remove(0);
           for cmd in cmds {
               state.command_queue.push_back(cmd);
           }
           dispatch_command(state, first, metro_tx);
       }
   }
   ```

4. Wire `Action::SyncBeforeRunAccept`:
   ```rust
   Action::SyncBeforeRunAccept => {
       if let Some(ModalState::SyncBeforeRun { run_command, needs_pods }) = state.modal.take() {
           // Enqueue: yarn install, (pod-install if needs_pods), then the run command
           state.command_queue.push_back(*run_command);
           if needs_pods {
               // Pod-install before run command but after yarn install
               // Re-order: yarn install first, pod-install, then run
               let run = state.command_queue.pop_back().unwrap();
               state.command_queue.push_back(CommandSpec::YarnPodInstall);
               state.command_queue.push_back(run);
           }
           dispatch_command(state, CommandSpec::YarnInstall, metro_tx);
       }
   }
   ```

```rust
fn render_clean_modal(f: &mut Frame, options: &crate::domain::command::CleanOptions) {
    let area = centered_rect(50, 12, f.area());

    // Clear background
    f.render_widget(ratatui::widgets::Clear, area);

    let checkbox = |checked: bool| if checked { "[x]" } else { "[ ]" };

    let text = vec![
        Line::from(Span::styled(" Clean Options ", Style::default().add_modifier(Modifier::BOLD))),
        Line::from(""),
        Line::from(format!("  n  {} node_modules (rm -rf)", checkbox(options.node_modules))),
        Line::from(format!("  p  {} CocoaPods (react-native clean)", checkbox(options.pods))),
        Line::from(format!("  a  {} Android (react-native clean)", checkbox(options.android))),
        Line::from(format!("  i  {} Sync after clean (yarn + pods)", checkbox(options.sync_after))),
        Line::from(""),
        Line::from(Span::styled("  x/Enter = confirm  Esc = cancel", Style::default().fg(Color::DarkGray))),
    ];

    let block = Block::default()
        .borders(Borders::ALL)
        .border_style(Style::default().fg(Color::Cyan))
        .title(" Clean ");

    let paragraph = Paragraph::new(text).block(block);
    f.render_widget(paragraph, area);
}
```

```rust
fn render_sync_prompt(f: &mut Frame, run_command: &crate::domain::command::CommandSpec, needs_pods: bool) {
    let area = centered_rect(50, 8, f.area());

    f.render_widget(ratatui::widgets::Clear, area);

    let sync_desc = if needs_pods {
        "yarn install + pod-install"
    } else {
        "yarn install"
    };

    let text = vec![
        Line::from(Span::styled(" Stale Dependencies ", Style::default().add_modifier(Modifier::BOLD))),
        Line::from(""),
        Line::from(format!("  Worktree is stale. Sync before running?")),
        Line::from(format!("  Will run: {} → {}", sync_desc, run_command.label())),
        Line::from(""),
        Line::from(Span::styled("  Y = sync first  N = skip  Esc = cancel", Style::default().fg(Color::DarkGray))),
    ];

    let block = Block::default()
        .borders(Borders::ALL)
        .border_style(Style::default().fg(Color::Yellow))
        .title(" Sync ");

    let paragraph = Paragraph::new(text).block(block);
    f.render_widget(paragraph, area);
}
```
